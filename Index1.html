// List your options here const options = [ "Order custom t-shirts from Wordans.ca", "Try a new restaurant today", "Go for a walk", "Watch a new movie tonight", "Start a new project" ]; // Function to randomly choose an option function chooseForMe(choices) { const randomIndex = Math.floor(Math.random() * choices.length); return choices[randomIndex]; } // Example usage: console.log("You should:", chooseForMe(options));
